Code Sample: Calculating Quarterly Growth


import pandas as pd

# Quarterly revenue data (in crores)
data = {
    "quarter": ["Q1", "Q2", "Q3", "Q4"],
    "revenue": [120.5, 135.2, 150.8, 160.3]
}

df = pd.DataFrame(data)

# Compute quarter-over-quarter growth (%)
df["qoq_growth_percent"] = df["revenue"].pct_change() * 100

print(df)
